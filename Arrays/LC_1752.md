class Solution {
public:
    bool check(vector<int>& nums) {
       int count =0;
       int n = nums.size()-1;
       if(nums[0]<nums[n]) ++count;

       for(int i=0;i<n;++i){
        if(nums[i]>nums[i+1]) ++count ;
       }

       if(count <2) return true ;
      
        return false ;
