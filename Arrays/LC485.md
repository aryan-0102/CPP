## 💻 Code Implementation

``````cpp
class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int max_count = 0; int count = 0 ;
        for(auto a:nums){
            max_count = max((a?++count : count =0 ),max_count);
        }
        return max_count ;
    }
};
